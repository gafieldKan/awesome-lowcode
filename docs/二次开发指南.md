## 开发一个自定义输入组件

### 场景

需要创建一个带字数统计和格式化的输入框组件。

### 步骤

**1. 创建插件文件**

在 `lowcode/plugins/my-plugins/` 目录下创建 `my-custom-input/index.js`：

```js
import { registerPlugin, PluginType } from '../../core/PluginRegistry'
import { componentEngine, ComponentCategory } from '../../engine/ComponentEngine'

// 注册插件
registerPlugin({
  name: 'my-custom-input',
  type: PluginType.FORM_COMPONENT,
  version: '1.0.0',
  description: '带字数统计的自定义输入框',

  async init() {
    // 注册到组件引擎
    componentEngine.registerComponent({
      id: 'customInput',
      name: '高级输入框',
      type: 'CustomInput',
      category: ComponentCategory.FORM,
      icon: 'InputOutlined',
      description: '带字数统计和格式校验的输入框',

      // 组件属性定义
      props: {
        placeholder: { type: 'string', default: '请输入' },
        maxLength: { type: 'number', default: 100 },
        format: { 
          type: 'select', 
          options: ['none', 'uppercase', 'lowercase', 'trim'], 
          default: 'none' 
        },
        showCount: { type: 'boolean', default: true },
        required: { type: 'boolean', default: false },
      },

      // 默认属性值
      defaultProps: {
        placeholder: '请输入内容',
        maxLength: 100,
        format: 'none',
        showCount: true,
        required: false,
      },

      // 支持的事件
      events: ['onChange', 'onFocus', 'onBlur'],
    })
  },

  async onLoad(context) {
    console.log('CustomInput 加载', context)
  },

  async onSave(data) {
    console.log('CustomInput 保存', data)
  },

  async onDestroy() {
    console.log('CustomInput 销毁')
  },
})

// 导出 React 组件
export const CustomInput = ({ 
  value, 
  onChange, 
  placeholder, 
  maxLength, 
  format, 
  showCount 
}) => {
  const handleChange = (e) => {
    let newValue = e.target.value

    // 格式化处理
    if (format === 'uppercase') {
      newValue = newValue.toUpperCase()
    } else if (format === 'lowercase') {
      newValue = newValue.toLowerCase()
    } else if (format === 'trim') {
      newValue = newValue.trim()
    }

    onChange?.(newValue)
  }

  return (
    <div style={{ position: 'relative' }}>
      <input
        type="text"
        value={value}
        onChange={handleChange}
        placeholder={placeholder}
        maxLength={maxLength}
        style={{
          width: '100%',
          padding: '8px 12px',
          border: '1px solid #d9d9d9',
          borderRadius: '4px',
          fontSize: '14px',
        }}
      />
      {showCount && (
        <span style={{
          position: 'absolute',
          right: 8,
          bottom: -20,
          fontSize: 12,
          color: '#999',
        }}>
          {value?.length || 0}/{maxLength}
        </span>
      )}
    </div>
  )
}

export default CustomInput
```

**2. 在应用中引入插件**

```js
import { registerPlugin } from 'lowcode/core/PluginRegistry'
import myCustomInput from './my-plugins/my-custom-input'

// 注册插件
registerPlugin(myCustomInput)
```

## 开发一个数据源插件

### 场景

需要对接外部 REST API 获取数据。

### 步骤

**创建 API 数据源插件**

```js
import { registerPlugin, PluginType } from '../../core/PluginRegistry'

registerPlugin({
  name: 'rest-api-datasource',
  type: PluginType.DATA_SOURCE,
  version: '1.0.0',
  description: 'REST API 数据源连接器',

  config: {
    timeout: 30000,
    retry: 3,
  },

  async init() {
    this.connections = new Map()
  },

  /**
   * 配置数据源连接
   */
  configure(name, config) {
    this.connections.set(name, {
      baseUrl: config.baseUrl,
      headers: config.headers || {},
      timeout: config.timeout || 30000,
    })
  },

  /**
   * GET 请求
   */
  async get(dataSourceName, endpoint, params = {}) {
    const config = this.connections.get(dataSourceName)
    if (!config) {
      throw new Error(`数据源未配置：${dataSourceName}`)
    }

    const url = new URL(endpoint, config.baseUrl)
    Object.entries(params).forEach(([key, value]) => {
      url.searchParams.append(key, value)
    })

    const response = await fetch(url, {
      method: 'GET',
      headers: config.headers,
      timeout: config.timeout,
    })

    if (!response.ok) {
      throw new Error(`HTTP Error: ${response.status}`)
    }

    return response.json()
  },

  /**
   * POST 请求
   */
  async post(dataSourceName, endpoint, data) {
    const config = this.connections.get(dataSourceName)
    if (!config) {
      throw new Error(`数据源未配置：${dataSourceName}`)
    }

    const url = new URL(endpoint, config.baseUrl)

    const response = await fetch(url, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        ...config.headers,
      },
      body: JSON.stringify(data),
      timeout: config.timeout,
    })

    if (!response.ok) {
      throw new Error(`HTTP Error: ${response.status}`)
    }

    return response.json()
  },

  async onDestroy() {
    this.connections?.clear()
  },
})
```

**使用示例**

```js
import { pluginRegistry } from 'lowcode/core/PluginRegistry'

// 获取插件实例
const apiPlugin = pluginRegistry.get('rest-api-datasource')

// 配置数据源
apiPlugin.configure('myApi', {
  baseUrl: 'https://api.example.com',
  headers: {
    'Authorization': 'Bearer token',
  },
})

// 获取数据
const users = await apiPlugin.get('myApi', '/users', { page: 1, size: 10 })

// 提交数据
const result = await apiPlugin.post('myApi', '/users', { name: 'John' })
```

## 开发一个工作流节点

### 场景

需要添加一个发送邮件通知的工作流节点。

### 步骤

```js
import { registerPlugin, PluginType } from '../../core/PluginRegistry'

registerPlugin({
  name: 'email-notification-node',
  type: PluginType.WORKFLOW_NODE,
  version: '1.0.0',
  description: '邮件通知节点',

  config: {
    smtpHost: '',
    smtpPort: 587,
  },

  async init() {
    console.log('邮件通知节点初始化')
  },

  /**
   * 获取节点配置
   */
  getNodeConfig() {
    return {
      type: 'email',
      label: '邮件通知',
      icon: 'mail',
      form: {
        recipients: [],
        subject: '',
        body: '',
        sendTime: 'immediate',
      },
    }
  },

  /**
   * 执行节点
   */
  async execute(context) {
    const { recipients, subject, body } = context

    console.log(`发送邮件给：${recipients.join(', ')}`)
    console.log(`主题：${subject}`)
    console.log(`内容：${body}`)

    // 实际实现应该调用邮件发送服务
    // await sendEmail({ to: recipients, subject, body })

    return {
      success: true,
      sentAt: Date.now(),
      recipientCount: recipients.length,
    }
  },

  /**
   * 验证配置
   */
  validateConfig(config) {
    const errors = []

    if (!config.recipients || config.recipients.length === 0) {
      errors.push('至少需要一个收件人')
    }

    if (!config.subject) {
      errors.push('邮件主题不能为空')
    }

    return errors
  },

  async onDestroy() {
    console.log('邮件通知节点销毁')
  },
})
```

## 权限集成示例

### 场景

需要根据用户角色控制表单字段的可见性和可编辑性。

### 实现

```js
import { permissionManager, PermissionLevel } from 'lowcode/core/PermissionManager'

// 1. 定义角色
permissionManager.registerRole({
  id: 'manager',
  name: '部门经理',
  permissions: [],
})

// 2. 设置模型权限
permissionManager.setModelPermission('employee', 'manager', PermissionLevel.WRITE)
permissionManager.setModelPermission('employee', 'user', PermissionLevel.READ)

// 3. 设置字段权限（薪资字段对普通用户不可见）
permissionManager.setFieldPermission('employee', 'salary', 'user', {
  visible: false,
  editable: false,
})

// 4. 设置字段权限（薪资字段对经理可见）
permissionManager.setFieldPermission('employee', 'salary', 'manager', {
  visible: true,
  editable: true,
})

// 5. 在组件中使用
function EmployeeForm({ userId }) {
  const role = permissionManager.getUserRole(userId)
  const salaryPerm = permissionManager.getFieldPermission('employee', 'salary', role?.id)

  return (
    <div>
      <input name="name" />
      {salaryPerm.visible && (
        <input name="salary" disabled={!salaryPerm.editable} />
      )}
    </div>
  )
}
```

## 部署和发布

### 1. 构建生产版本

```bash
npm run build
```

### 2. 预览构建结果

```bash
npm run preview
```

### 3. 发布插件

如果你开发了一个通用插件，可以发布到 npm：

```bash
# 在插件目录
cd lowcode/plugins/my-plugin
npm publish
```

### 4. 部署到服务器

```bash
# 构建
npm run build

# 将 dist 目录部署到 Web 服务器
# 或使用 Docker 部署
docker build -t awesome-lowcode .
docker run -p 80:80 awesome-lowcode
```

## 常见问题

### Q: 如何调试插件？

A: 在开发模式下，插件的方法会被调用，可以在控制台看到日志。也可以使用 `console.log` 或断点调试。

### Q: 插件可以依赖其他插件吗？

A: 可以。在插件配置中声明 `dependencies` 数组，系统会在加载时检查依赖。

### Q: 如何发布自定义主题？

A: 主题是一种特殊的插件，可以通过 CSS 变量和配置来定制 UI。
