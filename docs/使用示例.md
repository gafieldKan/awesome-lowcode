## 1. 事件引擎使用示例

### 1.1 基础事件订阅

```js
import { eventEngine, EventPriority, emit, on } from 'lowcode/engine/EventEngine'

// 方式一：使用 eventEngine 对象
eventEngine.on('user:login', (user) => {
  console.log('用户登录:', user.name)
})

// 方式二：使用快捷方法
on('data:create', (data) => {
  console.log('数据创建:', data)
})

// 发布事件
await emit('user:login', { id: 1, name: '张三' })
```

### 1.2 带优先级的订阅

```js
import { EventPriority } from 'lowcode/engine/EventEngine'

// 高优先级处理器（先执行）
eventEngine.on('order:submit', async (order) => {
  console.log('【高优】订单提交检查', order.id)
  // 库存检查
}, EventPriority.HIGH)

// 普通优先级
eventEngine.on('order:submit', async (order) => {
  console.log('订单提交通知', order.id)
  // 发送通知
}, EventPriority.NORMAL)

// 低优先级（后执行）
eventEngine.on('order:submit', async (order) => {
  console.log('订单提交日志', order.id)
  // 记录日志
}, EventPriority.LOW)
```

### 1.3 事件拦截器

```js
import { addInterceptor } from 'lowcode/engine/EventEngine'

// 添加拦截器，返回 false 可阻止后续执行
addInterceptor('data:delete', (data) => {
  const user = getCurrentUser()
  if (!user.isAdmin) {
    console.warn('无权限删除')
    return false // 阻止删除
  }
})

// 添加审计拦截器
addInterceptor('data:update', (data) => {
  auditLog.log('数据更新', {
    model: data.model,
    id: data.id,
    user: getCurrentUser()?.id,
    time: Date.now(),
  })
})
```

### 1.4 前后钩子

```js
import { eventEngine } from 'lowcode/engine/EventEngine'

// 添加前置钩子
eventEngine.addHook('app:start', async (event) => {
  console.log('应用启动前：加载配置...')
  await loadConfig()
})

// 添加后置钩子
eventEngine.addHook('app:start', async (event) => {
  console.log('应用启动后：初始化插件...')
  await initPlugins()
}, 'after')

// 移除钩子
eventEngine.removeHook('app:start', hookFn, 'before')
```

### 1.5 事件历史追踪

```js
import { eventEngine } from 'lowcode/engine/EventEngine'

// 启用调试模式
eventEngine.setDebugMode(true)

// 查看事件历史
const history = eventEngine.getHistory('data:update')
console.log('更新历史:', history)

// 清空历史
eventEngine.clearHistory()
```

## 2. 数据模型使用示例

### 2.1 定义模型

```js
import { defineModel, defineField, FieldType, ModelConfig } from 'lowcode/core/ModelSchema'

// 定义用户模型
const userSchema = defineModel({
  id: 'user',
  name: 'User',
  label: '用户',
  description: '系统用户模型',
  fields: [
    defineField({
      name: 'username',
      label: '用户名',
      type: FieldType.STRING,
      required: true,
      unique: true,
      validation: [
        { type: 'minLength', value: 3, message: '用户名至少 3 个字符' },
        { type: 'maxLength', value: 20, message: '用户名最多 20 个字符' },
        { type: 'pattern', pattern: '^[a-zA-Z0-9_]+$', message: '只能包含字母、数字和下划线' },
      ],
    }),
    defineField({
      name: 'email',
      label: '邮箱',
      type: FieldType.EMAIL,
      required: true,
      validation: [{ type: 'email' }],
    }),
    defineField({
      name: 'phone',
      label: '手机号',
      type: FieldType.PHONE,
      validation: [{ type: 'phone' }],
    }),
    defineField({
      name: 'age',
      label: '年龄',
      type: FieldType.INTEGER,
      validation: [
        { type: 'min', value: 18, message: '年龄不能小于 18' },
        { type: 'max', value: 100, message: '年龄不能大于 100' },
      ],
    }),
    defineField({
      name: 'status',
      label: '状态',
      type: FieldType.SELECT,
      options: [
        { label: '激活', value: 'active' },
        { label: '禁用', value: 'disabled' },
        { label: '待审核', value: 'pending' },
      ],
      default: 'pending',
    }),
  ],
})
```

### 2.2 注册到 Schema 管理器

```js
import { schemaManager } from 'lowcode/core/SchemaManager'

// 注册模型
schemaManager.registerModel('user', userSchema)

// 获取模型
const userModel = schemaManager.getModel('user')

// 验证数据
const testData = { username: 'zhang', email: 'test@example.com', age: 25 }
const errors = userModel.validate(testData)
if (errors.length > 0) {
  console.log('验证失败:', errors)
}
```

### 2.3 自定义验证规则

```js
import { FieldConfig } from 'lowcode/core/ModelSchema'

const customField = new FieldConfig({
  name: 'password',
  label: '密码',
  type: 'string',
  required: true,
  validation: [
    {
      type: 'custom',
      validate: (value, data) => {
        if (value.length < 8) {
          return '密码长度至少 8 位'
        }
        if (!/[A-Z]/.test(value)) {
          return '密码必须包含大写字母'
        }
        if (!/[0-9]/.test(value)) {
          return '密码必须包含数字'
        }
        return null // 验证通过
      },
    },
  ],
})
```

### 2.4 导出 JSON Schema

```js
// 转换为标准 JSON Schema
const jsonSchema = userSchema.toJSONSchema()
console.log(JSON.stringify(jsonSchema, null, 2))

// 输出：
// {
//   "$schema": "http://json-schema.org/draft-en/schema#",
//   "title": "用户",
//   "type": "object",
//   "properties": { ... },
//   "required": ["username", "email"]
// }
```

## 3. 权限管理使用示例

### 3.1 基础权限配置

```js
import {
  permissionManager,
  Operations,
  DataScope,
} from 'lowcode/core/PermissionManager'

// 1. 注册新角色
permissionManager.registerRole({
  id: 'manager',
  name: '部门经理',
  description: '部门管理人员',
})

// 2. 为用户分配角色
permissionManager.assignRole('user123', 'manager')
permissionManager.assignRole('user123', 'editor') // 支持多角色

// 3. 设置模型权限
permissionManager.setModelPermission('order', 'manager', [
  Operations.READ,
  Operations.WRITE,
  Operations.CREATE,
  Operations.DELETE,
])

// 4. 检查权限
const canRead = permissionManager.hasModelPermission('order', 'manager', Operations.READ)
console.log('是否可以读取:', canRead) // true
```

### 3.2 字段级权限

```js
import { permissionManager } from 'lowcode/core/PermissionManager'

// 设置字段权限
permissionManager.setFieldPermission('user', 'salary', 'editor', {
  visible: false,
  readable: false,
  writable: false,
})

permissionManager.setFieldPermission('user', 'salary', 'manager', {
  visible: true,
  readable: true,
  writable: true,
})

// 检查字段权限
const fieldPerm = permissionManager.getFieldPermission('user', 'salary', 'editor')
console.log('薪资字段可见性:', fieldPerm.visible) // false

// 快捷方法
const isWritable = permissionManager.isFieldWritable('user', 'salary', 'manager')
console.log('经理可编辑薪资:', isWritable) // true
```

### 3.3 数据权限（行级权限）

```js
import { permissionManager, DataScope } from 'lowcode/core/PermissionManager'

// 设置数据范围
permissionManager.setDataScope('order', 'sales', DataScope.MINE) // 仅本人数据
permissionManager.setDataScope('order', 'manager', DataScope.DEPT) // 本部门数据
permissionManager.setDataScope('order', 'director', DataScope.ALL) // 全部数据

// 构建数据过滤条件
const filter = permissionManager.buildDataFilter('order', 'user123')
console.log('数据过滤条件:', filter)
// 输出: { createBy: 'user123' } (对于 MINE 范围)
```

### 3.4 批量设置权限

```js
import { permissionManager, Operations } from 'lowcode/core/PermissionManager'

permissionManager.batchSetPermissions([
  {
    type: 'model',
    modelId: 'order',
    role: 'sales',
    operations: [Operations.READ, Operations.CREATE],
  },
  {
    type: 'model',
    modelId: 'order',
    role: 'manager',
    operations: [Operations.READ, Operations.WRITE, Operations.DELETE],
  },
  {
    type: 'field',
    modelId: 'user',
    field: 'salary',
    role: 'sales',
    visible: false,
    writable: false,
  },
  {
    type: 'action',
    action: 'export',
    role: 'guest',
    allowed: false,
  },
  {
    type: 'menu',
    menu: 'settings',
    role: 'guest',
    visible: false,
  },
])
```

### 3.5 获取用户权限摘要

```js
import { getUserPermissions } from 'lowcode/core/PermissionManager'

// 获取用户对模型的所有权限
const permissions = getUserPermissions('user123', 'order')
console.log('用户权限:', permissions)
// 输出:
// {
//   canRead: true,
//   canWrite: false,
//   canCreate: true,
//   canDelete: false,
//   canSubmit: false,
//   canApprove: false,
// }
```

## 4. 综合应用示例

### 4.1 订单审批流程

```js
import { eventEngine, emit } from 'lowcode/engine/EventEngine'
import { permissionManager, Operations, DataScope } from 'lowcode/core/PermissionManager'
import { defineModel, defineField, FieldType } from 'lowcode/core/ModelSchema'

// 1. 定义订单模型
const orderSchema = defineModel({
  id: 'order',
  name: 'Order',
  label: '订单',
  fields: [
    defineField({ name: 'orderNo', label: '订单号', type: FieldType.STRING, required: true }),
    defineField({ name: 'amount', label: '金额', type: FieldType.CURRENCY, required: true }),
    defineField({ name: 'status', label: '状态', type: FieldType.SELECT, options: [
      { label: '草稿', value: 'draft' },
      { label: '待审批', value: 'pending' },
      { label: '已批准', value: 'approved' },
      { label: '已拒绝', value: 'rejected' },
    ]}),
  ],
})

// 2. 设置权限
permissionManager.setModelPermission('order', 'sales', [Operations.READ, Operations.CREATE])
permissionManager.setModelPermission('order', 'manager', [Operations.READ, Operations.APPROVE])
permissionManager.setDataScope('order', 'sales', DataScope.MINE)
permissionManager.setDataScope('order', 'manager', DataScope.ALL)

// 3. 注册审批事件处理器
eventEngine.on('order:submit', async (order) => {
  const user = getCurrentUser()
  if (!permissionManager.hasModelPermission('order', user.role, Operations.SUBMIT)) {
    throw new Error('无提交权限')
  }
  order.status = 'pending'
  await saveOrder(order)
}, EventPriority.HIGH)

eventEngine.on('order:approve', async (order) => {
  const user = getCurrentUser()
  if (!permissionManager.hasModelPermission('order', user.role, Operations.APPROVE)) {
    throw new Error('无审批权限')
  }
  // 审批逻辑
}, EventPriority.NORMAL)

// 4. 添加审批前后钩子
eventEngine.addHook('order:approve', async (event, order) => {
  console.log(`订单 ${order.id} 审批前检查...`)
  if (order.amount > 100000) {
    throw new Error('大额订单需要更高级别审批')
  }
}, 'before')

eventEngine.addHook('order:approve', async (event, order) => {
  // 发送审批通过通知
  sendNotification(order.createBy, '订单审批通过')
}, 'after')

// 5. 触事件
await emit('order:submit', { id: 1, orderNo: 'ORD001', amount: 5000 })
```

### 4.2 完整的数据访问控制

```js
import { eventEngine, emit, addInterceptor } from 'lowcode/engine/EventEngine'
import { permissionManager, DataScope } from 'lowcode/core/PermissionManager'

// 添加全局数据访问拦截器
addInterceptor('data:read', async (context) => {
  const { model, data, user } = context

  // 获取用户的数据权限范围
  const dataScope = permissionManager.getDataScope(model, user.role)

  // 根据数据范围构建过滤条件
  let filter = {}
  switch (dataScope.scope) {
    case DataScope.MINE:
      filter = { createBy: user.id }
      break
    case DataScope.DEPT:
      filter = { departmentId: user.departmentId }
      break
    case DataScope.ALL:
      // 无需过滤
      break
    default:
      filter = { createBy: user.id }
  }

  // 将过滤条件应用到查询
  context.filter = filter
})

// 添加字段可见性拦截
addInterceptor('field:read', async (context) => {
  const { model, field, user } = context
  const perm = permissionManager.getFieldPermission(model, field, user.role)

  if (!perm.visible) {
    context.skip = true
  }
})
```

## 5. 最佳实践建议

### 5.1 事件命名规范

```js
// 推荐格式：{模块}:{操作}
'user:login'      // ✅ 好
'user_logout'     // ❌ 不推荐
'dataCreated'     // ❌ 不推荐

// 标准事件类型
'app:*'     // 应用事件
'data:*'    // 数据事件
'user:*'    // 用户事件
'plugin:*'  // 插件事件
'component:*' // 组件事件
```

### 5.2 权限设计原则

1. **最小权限原则** - 默认拒绝，按需授权
2. **角色分离** - 不同角色职责分离
3. **数据隔离** - 使用数据范围隔离数据
4. **审计追踪** - 关键操作记录日志

### 5.3 性能优化

```js
// 使用事件缓存
const checker = permissionManager.getPermissionChecker(userId)

// 批量设置权限，减少缓存失效
permissionManager.batchSetPermissions([...])

// 及时清理不再使用的事件订阅
eventEngine.off(eventId, callbackId)
```
