### 核心 API

```js
import { eventEngine, EventPriority } from 'lowcode/engine/EventEngine'

// 订阅事件（带优先级）
eventEngine.on('order:submit', handler, EventPriority.HIGH)

// 添加拦截器
addInterceptor('data:delete', (data) => {
  if (!canDelete) return false // 阻止删除
})

// 添加钩子
eventEngine.addHook('app:start', asyncFn, 'before')
eventEngine.addHook('app:start', asyncFn, 'after')

// 调试模式
eventEngine.setDebugMode(true)
```


### 字段类型扩展

| 类型分类 | 新增类型 |
|----------|----------|

| 日期时间 | `date`, `datetime`, `time` |
| 选择类型 | `select`, `multiSelect`, `radio`, `checkbox`, `tag` |
| 文件类型 | `image`, `file`, `images`, `files` |
| 高级类型 | `json`, `code`, `color`, `rating`, `percent`, `currency`, `phone`, `email`, `url`, `address`, `formula`, `autoNumber` |

### 验证规则

| 规则类型 | 说明 |
|----------|------|
| `required` | 必填 |
| `pattern` | 正则匹配 |
| `min`/`max` | 最小/最大值 |
| `minLength`/`maxLength` | 长度限制 |
| `email` | 邮箱格式 |
| `phone` | 手机号格式 |
| `url` | URL 格式 |
| `number`/`integer` | 数字/整数 |
| `custom` | 自定义验证 |

### 核心 API

```js
import { defineModel, defineField, FieldType } from 'lowcode/core/ModelSchema'

// 定义模型
const userSchema = defineModel({
  id: 'user',
  name: 'User',
  label: '用户',
  fields: [
    defineField({
      name: 'username',
      label: '用户名',
      type: FieldType.STRING,
      required: true,
      unique: true,
      validation: [
        { type: 'minLength', value: 3 },
        { type: 'maxLength', value: 20 },
        { type: 'pattern', pattern: '^[a-zA-Z0-9]+$' },
      ],
    }),
    defineField({
      name: 'email',
      label: '邮箱',
      type: FieldType.EMAIL,
      validation: [{ type: 'email' }],
    }),
  ],
})

// 验证数据
const errors = userSchema.validate({ username: 'ab', email: 'invalid' })
// 输出：[{ field: 'username', errors: ['长度不能小于 3'] }, ...]

// 导出 JSON Schema
const jsonSchema = userSchema.toJSONSchema()
```

| 批量授权 | 批量设置权限 | ✅ 完成 |
| 权限缓存 | 权限检查缓存优化 | ✅ 完成 |

### 操作类型（14 种）

```js
export const Operations = {
  READ: 'read',
  WRITE: 'write',
  CREATE: 'create',
  DELETE: 'delete',
  UPDATE: 'update',
  SUBMIT: 'submit',      // 提交审批
  CANCEL: 'cancel',      // 取消/作废
  APPROVE: 'approve',    // 审批
  REJECT: 'reject',      // 拒绝
  SHARE: 'share',        // 分享
  EXPORT: 'export',      // 导出
  IMPORT: 'import',      // 导入
  PRINT: 'print',        // 打印
  FULL: 'full',          // 完全控制
}
```

### 数据范围

```js
export const DataScope = {
  ALL: 'all',              // 全部数据
  MINE: 'mine',            // 仅本人数据
  DEPT: 'dept',            // 本部门数据
  DEPT_AND_SUB: 'dept_and_sub',  // 本部门及子部门数据
  CUSTOM: 'custom',        // 自定义范围
}
```

### 核心 API

```js
import {
  permissionManager,
  Operations,
  DataScope,
} from 'lowcode/core/PermissionManager'

// 1. 分配角色
permissionManager.assignRole('user123', 'manager')
permissionManager.assignRole('user123', 'editor')

// 2. 设置模型权限
permissionManager.setModelPermission('order', 'manager', [
  Operations.READ,
  Operations.WRITE,
  Operations.APPROVE,
])

// 3. 设置数据范围
permissionManager.setDataScope('order', 'manager', DataScope.DEPT)

// 4. 设置字段权限
permissionManager.setFieldPermission('user', 'salary', 'editor', {
  visible: false,
  readable: false,
  writable: false,
})

// 5. 检查权限
const canApprove = permissionManager.hasModelPermission(
  'order',
  'manager',
  Operations.APPROVE
)

// 6. 批量授权
permissionManager.batchSetPermissions([
  { type: 'model', modelId: 'order', role: 'sales', operations: [...] },
  { type: 'field', modelId: 'user', field: 'salary', role: 'sales', visible: false },
])

// 7. 获取用户权限摘要
const permissions = permissionManager.getUserModelPermissions('user123', 'order')
```
